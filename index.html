obstacle_navigation.py Complete Implementation 
import random
import time

# ================ GRID ENVIRONMENT CLASS ================
class Grid:
    def __init__(self, size, obstacles):
        self.size = size        # Encapsulation: size is private
        self.obstacles = obstacles

    def is_obstacle(self, x, y):
        # Check boundaries and obstacles
        return (x, y) in self.obstacles or x < 0 or y < 0 or x >= self.size or y >= self.size

    def display(self, robot):
        # Display grid with robot and obstacles
        for y in range(self.size - 1, -1, -1):
            for x in range(self.size):
                if robot.x == x and robot.y == y:
                    print(" R ", end="")
                elif (x, y) in self.obstacles:
                    print(" X ", end="")
                else:
                    print(" . ", end="")
            print()
        print("-" * 20)

# ================ SENSOR CLASS ================
class ObstacleSensor:
    def detect(self, grid, x, y):
        return grid.is_obstacle(x, y)

# ================ ROBOT CLASS ================
class Robot:
    def __init__(self, x, y):
        self.x = x              # Encapsulated position
        self.y = y
        self.direction = random.choice(["up", "down", "left", "right"])

    def next_position(self):
        if self.direction == "up":
            return self.x, self.y + 1
        elif self.direction == "down":
            return self.x, self.y - 1
        elif self.direction == "left":
            return self.x - 1, self.y
        else:
            return self.x + 1, self.y

    def move(self):
        nx, ny = self.next_position()
        self.x, self.y = nx, ny

    def change_direction(self):
        self.direction = random.choice(["up", "down", "left", "right"])

# ================ CONTROLLER CLASS ================
class Controller:
    def __init__(self, robot, sensor, grid):
        self.robot = robot
        self.sensor = sensor
        self.grid = grid

    def navigate(self):
        nx, ny = self.robot.next_position()
        if self.sensor.detect(self.grid, nx, ny):
            print("⚠ Obstacle detected → Changing direction")
            self.robot.change_direction()
        else:
            self.robot.move()
            print(f"✅ Robot moved {self.robot.direction}")

# ================ MAIN PROGRAM ================
# Object Creation and Interaction
grid = Grid(5, obstacles=[(1, 1), (2, 2), (3, 1), (3, 3), (4, 0)])
robot = Robot(0, 0)
sensor = ObstacleSensor()
controller = Controller(robot, sensor, grid)

# Simulation Loop (20 steps)
for step in range(20):
    print(f"\nStep {step + 1}")
    controller.navigate()
    grid.display(robot)
    time.sleep(1)
Navigation Logic

Working Algorithm

Step-by-step process of how the robot makes intelligent navigation decisions

1

Sense the Environment

Robot uses its sensor to detect obstacles in the next position before moving

2

Decision Making

Controller analyzes sensor data and decides the appropriate movement strategy

3

Obstacle Avoidance

If obstacle detected, robot changes direction randomly to find a clear path

4

Movement Execution

If path is clear, robot moves forward in the current direction

5

Continuous Loop

Process repeats for 20 steps, creating a complete navigation simulation
